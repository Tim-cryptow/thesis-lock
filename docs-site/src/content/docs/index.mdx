import { Card, CardGrid, LinkCard } from "@astrojs/starlight/components";

ThesisLock is a hash-anchoring service for academic and creative documents. You drop a
file into the web app, the browser hashes it with SHA-256, and you sign a Stacks
transaction that records the hash on chain with an optional label. Anyone can later
verify when the anchor was created, by which wallet, and what label was attached, then
re-upload a file to confirm its hash matches. The result is a permanent, publicly
verifiable timestamp that never exposes the document itself.

<CardGrid>
  <Card title="Private by design" icon="approve-check">
    Your file is hashed in the browser. Only the 64-character digest is published on
    chain, never the document.
  </Card>
  <Card title="Permanent and verifiable" icon="star">
    Once anchored, a hash and its timestamp are immutable. Verification re-hashes your
    file locally and checks it against the chain.
  </Card>
  <Card title="Secured by Bitcoin" icon="rocket">
    Stacks settles to Bitcoin, so each anchor records the Bitcoin (burn) block height it
    was confirmed against.
  </Card>
  <Card title="Open and scriptable" icon="puzzle">
    Verify anchors from a TypeScript SDK, a terminal CLI, a GitHub Action, or the public
    REST API.
  </Card>
</CardGrid>
